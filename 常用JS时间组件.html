			const pastTimeFn =(timestamp)=>{
				var nowDate = new Date();
				var nowTimestamp = nowDate.getTime();
				var timevalue = nowTimestamp - timestamp;
				var timeTip = '';

				if(timevalue < 60*1000){
					timeTip = parseInt(timevalue/(1000)) + '秒前';
				}else if(timevalue < 60*60*1000){
					timeTip = parseInt(timevalue/(60*1000)) + '分前';  
				}else if(timevalue < 24*60*60*1000){
					timeTip = parseInt(timevalue/(60*60*1000)) + '小时前';  
				}else if(timevalue < 2*24*60*60*1000){
					timeTip = '昨天';  
				}else if(timevalue < 30*24*60*60*1000){
					timeTip = parseInt(timevalue/(24*60*60*1000)) + '天前';  
				}else if(timevalue < 365*24*60*60*1000){
					timeTip =  getDate(parseInt(timestamp)).substring(5,16);
				}else{
					timeTip =  getDate(parseInt(timestamp)).substring(0,10);
				}

				return timeTip;
